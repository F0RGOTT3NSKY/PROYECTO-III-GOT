27 /* To add a new kind of warning, add an
28  * if(curlcheck_sometype_option(_curl_opt))
29  * if(!curlcheck_sometype(value))
30  * _curl_easy_setopt_err_sometype();
31  * block and define curlcheck_sometype_option, curlcheck_sometype and
32  * _curl_easy_setopt_err_sometype below
33  *
34  * NOTE: We use two nested 'if' statements here instead of the && operator, in
35  * order to work around gcc bug #32061. It affects only gcc 4.3.x/4.4.x
36  * when compiling with -Wlogical-op.
37  *
38  * To add an option that uses the same type as an existing option, you'll just
39  * need to extend the appropriate _curl_*_option macro
40  */

427 /* typecheck helpers -- check whether given expression has requested type*/
428 
429 /* For pointers, you can use the curlcheck_ptr/curlcheck_arr macros,
430  * otherwise define a new macro. Search for __builtin_types_compatible_p
431  * in the GCC manual.
432  * NOTE: these macros MUST NOT EVALUATE their arguments! The argument is
433  * the actual expression passed to the curl_easy_setopt macro. This
434  * means that you can only apply the sizeof and __typeof__ operators, no
435  * == or whatsoever.
436  */
437 
438 /* XXX: should evaluate to true if expr is a pointer */
439 #define curlcheck_any_ptr(expr) \
440  (sizeof(expr) == sizeof(void *))
441 
442 /* evaluates to true if expr is NULL */
443 /* XXX: must not evaluate expr, so this check is not accurate */
444 #define curlcheck_NULL(expr) \
445  (__builtin_types_compatible_p(__typeof__(expr), __typeof__(NULL)))
446 
447 /* evaluates to true if expr is type*, const type* or NULL */
448 #define curlcheck_ptr(expr, type) \
449  (curlcheck_NULL(expr) || \
450  __builtin_types_compatible_p(__typeof__(expr), type *) || \
451  __builtin_types_compatible_p(__typeof__(expr), const type *))
452 
453 /* evaluates to true if expr is one of type[], type*, NULL or const type* */
454 #define curlcheck_arr(expr, type) \
455  (curlcheck_ptr((expr), type) || \
456  __builtin_types_compatible_p(__typeof__(expr), type []))
457 
458 /* evaluates to true if expr is a string */
459 #define curlcheck_string(expr) \

462  curlcheck_arr((expr), unsigned char))
